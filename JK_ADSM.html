### Set up ###

# Generate hospital index 
hosp <- read.csv("hosp_full.csv", header=TRUE)
hosp_table <- data.frame(id = unique(hosp$id), hosp_index = c(1:length(unique(hosp$id))))
hosp <- inner_join(hosp, hosp_table) %>%
  filter(category!="Overall")
## Joining, by = "id"
hosp_red <- read.csv("hosp.csv") # "reduced" hospital data


y = hosp$deaths
n = hosp$procedures
X = model.matrix(deaths ~ category, data=hosp)[,-1] # make category 1 baseline
Z =  model.matrix(deaths ~ log(procedures) + top50_ped_cardio + 
                   top50_ped_pulm , data=hosp_red)

g = hosp$hosp_index

N = nrow(hosp)
M = length(unique(hosp$id))
K = ncol(X)
P = ncol(Z)


### Actually Sampling ###
stan_data = list(N=N, M = M, K = K, P = P, n=n, y = y, x = X, z = Z, g = g)

options(mc.cores = 2)
start_time <- Sys.time()
fit1 <- stan("JK_ADSM.stan", data = stan_data, chains = 2, iter = 1000, cores = 2)
